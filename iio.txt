create table Groups (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    group_name varchar(50),
    kurs integer check (kurs > 0 and kurs < 4)    
);

create table Student (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    i_name varchar2(900),
    f_name varchar2(100),
    o_name varchar(100),
    sex varchar2(10),
    id_group references Groups(id)
);

create table DirGrades (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name_grade integer check(name_grade > 1 and name_grade < 6),
    discription varchar2(100)
);

create table Vedomost (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    semestr varchar2(50),
    kurs integer check (kurs > 0 and kurs < 4),
    id_group references Groups(id),
    teacher varchar(100),
    discription varchar2(100),
    hours_d integer check (hours_d > 50 and hours_d < 250)
);

create table Grades (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    student_id references Student(id),
    ved_id references Vedomost(id),
    id_grade references DirGrades(id) 
);



3. Функция (средний балл студента по всем предметам)
CREATE OR REPLACE FUNCTION GetStudentAvgGrade(p_student_id INTEGER)
RETURN NUMBER
IS
    v_avg NUMBER;
BEGIN
    SELECT AVG(dg.name_grade) INTO v_avg
    FROM Grades g
    JOIN DirGrades dg ON g.id_grade = dg.id
    WHERE g.student_id = p_student_id;
    
    RETURN NVL(v_avg, 0);
END GetStudentAvgGrade;
/


Пример вызова
SELECT GetStudentAvgGrade(1) FROM dual;










Триггер (trg_check_student_kurs)
Срабатывает перед вставкой или обновлением оценки. 
Проверяет, совпадает ли курс студента (через его группу) с курсом, указанным в ведомости. Если курсы разные — блокирует операцию и выдаёт ошибку. Обеспечивает целостность данных.







Функция (GetStudentAvgGrade)
Возвращает средний балл студента по всем предметам. 
Принимает ID студента, вычисляет среднее арифметическое его оценок из таблицы Grades. 
Если оценок нет — возвращает 0. Можно использовать в SELECT-запросах.


Представление (StudentGradesView)
Виртуальная таблица, объединяющая данные из 5 таблиц (Student, Groups, Grades, DirGrades, Vedomost). 
Показывает полное ФИО студента, группу, пол, оценку с описанием, семестр, преподавателя. 
Упрощает написание сложных отчётов — можно делать SELECT как с обычной таблицы.















-- ============================================================
-- ПРОЦЕДУРА: добавление студента с проверкой дубликатов
-- ============================================================
CREATE OR REPLACE PROCEDURE AddStudent(
    -- Входные параметры (IN - только для чтения внутри процедуры)
    p_i_name IN VARCHAR2,   -- фамилия
    p_f_name IN VARCHAR2,   -- имя
    p_o_name IN VARCHAR2,   -- отчество
    p_sex IN VARCHAR2,      -- пол
    p_group_id IN INTEGER   -- ID группы
)
IS
    -- Секция объявления локальных переменных
    v_exists INTEGER;
BEGIN
    -- ===== Проверка: есть ли уже студент с таким ФИО в этой группе? =====
    -- COUNT(*) подсчитает количество совпадающих строк
    SELECT COUNT(*) INTO v_exists
    FROM Student
    WHERE i_name = p_i_name 
      AND f_name = p_f_name 
      AND o_name = p_o_name
      AND id_group = p_group_id;
    
    -- ===== Логика выбора =====
    IF v_exists > 0 THEN
        -- Если дубликат найден – вызываем пользовательскую ошибку
        -- -20001 ... -20999 — диапазон пользовательских ошибок Oracle
        RAISE_APPLICATION_ERROR(-20001, 
            'Студент ' || p_i_name || ' ' || p_f_name || ' ' || p_o_name || 
            ' уже существует в группе ' || p_group_id);
    ELSE
        -- Вставка нового студента
        INSERT INTO Student (i_name, f_name, o_name, sex, id_group)
        VALUES (p_i_name, p_f_name, p_o_name, p_sex, p_group_id);
        
        -- Информационное сообщение (видно при SET SERVEROUTPUT ON)
        DBMS_OUTPUT.PUT_LINE('Студент успешно добавлен');
    END IF;
    
    -- Фиксация транзакции (сохраняем изменения)
    COMMIT;
EXCEPTION
    -- Блок обработки ошибок
    WHEN OTHERS THEN
        -- Откат любых изменений в рамках этой процедуры
        ROLLBACK;
        -- Переброс исключения наверх (чтобы вызывающий код знал об ошибке)
        RAISE;
END AddStudent;
/


SET SERVEROUTPUT ON;
BEGIN
    AddStudent('Иванов', 'Иван', 'Иванович', 'М', 1);
END;
/














-- ============================================================
-- ТРИГГЕР: проверка соответствия курса студента и курса ведомости
-- Срабатывает ПЕРЕД вставкой или обновлением записи в Grades
-- ============================================================
CREATE OR REPLACE TRIGGER check_grade_course_trigger
BEFORE INSERT OR UPDATE ON Grades   -- Событие: перед вставкой/обновлением
FOR EACH ROW                        -- Триггер уровня строки (для каждой строки)
DECLARE
    -- Локальные переменные для хранения курсов
    v_student_kurs INTEGER;
    v_vedomost_kurs INTEGER;
BEGIN
    -- :NEW – псевдозапись с новыми значениями (то, что пытаются вставить/обновить)
    -- :OLD – старые значения (для UPDATE)
    
    -- Получаем курс студента по его ID (из связанной таблицы Student)
    SELECT s.kurs INTO v_student_kurs
    FROM Student s
    WHERE s.id = :NEW.student_id;
    
    -- Получаем курс из ведомости по ID ведомости
    SELECT v.kurs INTO v_vedomost_kurs
    FROM Vedomost v
    WHERE v.id = :NEW.ved_id;
    
    -- Проверяем совпадение курсов
    IF v_student_kurs != v_vedomost_kurs THEN
        -- Если не совпадают – генерируем ошибку (операция INSERT/UPDATE отменится)
        RAISE_APPLICATION_ERROR(-20002, 
            'Ошибка: Курс студента (' || v_student_kurs || 
            ') не соответствует курсу ведомости (' || v_vedomost_kurs || ')');
    END IF;
    
    -- Если всё хорошо – триггер завершается, и INSERT/UPDATE выполняется
END;
/
