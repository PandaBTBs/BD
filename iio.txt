create table Groups (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    group_name varchar(50),
    kurs integer check (kurs > 0 and kurs < 4)    
);

create table Student (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    i_name varchar2(900),
    f_name varchar2(100),
    o_name varchar(100),
    sex varchar2(10),
    id_group references Groups(id)
);

create table DirGrades (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name_grade integer check(name_grade > 1 and name_grade < 6),
    discription varchar2(100)
);

create table Vedomost (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    semestr varchar2(50),
    kurs integer check (kurs > 0 and kurs < 4),
    id_group references Groups(id),
    teacher varchar(100),
    discription varchar2(100),
    hours_d integer check (hours_d > 50 and hours_d < 250)
);

create table Grades (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    student_id references Student(id),
    ved_id references Vedomost(id),
    id_grade references DirGrades(id) 
);



3. Функция (средний балл студента по всем предметам)
CREATE OR REPLACE FUNCTION GetStudentAvgGrade(p_student_id INTEGER)
RETURN NUMBER
IS
    v_avg NUMBER;
BEGIN
    SELECT AVG(dg.name_grade) INTO v_avg
    FROM Grades g
    JOIN DirGrades dg ON g.id_grade = dg.id
    WHERE g.student_id = p_student_id;
    
    RETURN NVL(v_avg, 0);
END GetStudentAvgGrade;
/


Пример вызова
SELECT GetStudentAvgGrade(1) FROM dual;










Триггер (trg_check_student_kurs)
Срабатывает перед вставкой или обновлением оценки. 
Проверяет, совпадает ли курс студента (через его группу) с курсом, указанным в ведомости. Если курсы разные — блокирует операцию и выдаёт ошибку. Обеспечивает целостность данных.







Функция (GetStudentAvgGrade)
Возвращает средний балл студента по всем предметам. 
Принимает ID студента, вычисляет среднее арифметическое его оценок из таблицы Grades. 
Если оценок нет — возвращает 0. Можно использовать в SELECT-запросах.


Представление (StudentGradesView)
Виртуальная таблица, объединяющая данные из 5 таблиц (Student, Groups, Grades, DirGrades, Vedomost). 
Показывает полное ФИО студента, группу, пол, оценку с описанием, семестр, преподавателя. 
Упрощает написание сложных отчётов — можно делать SELECT как с обычной таблицы.
